Etapa 5 — REMOVER NUVENS

Define uma função para mascarar nuvens, sombras e pixels indesejados do Sentinel-2.

def mask_sentinel2(image):                # Função para remover nuvens do Sentinel-2

    scl = image.select('SCL')             # Seleciona a banda de classificação da cena

    mask = (
        scl.neq(3)                        # Sombra de nuvem
        .And(scl.neq(8))                  # Nuvem média
        .And(scl.neq(9))                  # Nuvem alta
        .And(scl.neq(10))                 # Cirrus
        .And(scl.neq(11))                 # Neve/gelo
    )

    return (
        image
        .updateMask(mask)                 # Aplica a máscara
        .divide(10000)                    # Converte para reflectância
        .copyProperties(image, ['system:time_start'])  # Mantém a data da imagem
    )

Etapa 6 — CALCULAR INDICES (NDVI, NDWI, CMRI)

Cria as bandas NDVI, NDWI e CMRI e adiciona esses índices à imagem.

def add_indices(img):                           # Calcula os índices espectrais

    ndvi = img.normalizedDifference(['B8','B4']).rename('NDVI')  # NDVI
    ndwi = img.normalizedDifference(['B3','B8']).rename('NDWI')  # NDWI
    cmri = ndvi.subtract(ndwi).rename('CMRI')                    # CMRI

    return img.addBands([ndvi, ndwi, cmri])    # Adiciona os índices à imagem

Etapa 7 — CARREGA AS IMANGENS DO SENTINEL

Monta a coleção Sentinel-2 filtrada pela área, percentual de nuvens e índices calculados.

sen2_class = (
    ee.ImageCollection('COPERNICUS/S2_SR_HARMONIZED') # Coleção Sentinel-2
    .filterBounds(roi)                                # Filtra pela área de interesse
    .filter(ee.Filter.lt('CLOUDY_PIXEL_PERCENTAGE', 30)) # Filtra imagens com menos de 30% de nuvens
    .map(mask_sentinel2)                              # Remove nuvens e sombras
    .map(add_indices)                                 # Calcula NDVI, NDWI e CMRI
)

    )

    area = (
        ee.Image.pixelArea()                # Cria uma imagem contendo a área de cada pixel em m²
        .updateMask(cmri_bin)               # Mantém apenas os pixels classificados como manguezal
        .rename('area')                     # Renomeia a banda para "area"
        .reduceRegion(
            reducer=ee.Reducer.sum(),       # Soma a área dos pixels classificados
            geometry=roi,                   # Limita o cálculo à área de interesse
            scale=10,                       # Utiliza a resolução espacial do Sentinel-2 (10 m)
            maxPixels=1e13                  # Define o número máximo de pixels processados
        )
    )

    area_ha = (
        area
        .get('area')                        # Obtém a área total em metros quadrados
        .getInfo()
        / 10000                             # Converte a área para hectares
    )

    areas_ha.append(area_ha)                # Armazena a área correspondente ao ano analisado

    print(
        f'{year}: {area_ha:.2f} ha'
    )                                       # Exibe a área estimada para cada ano
