Etapa 5 — REMOVER NUVENS

Define a função de máscara para remover sombras, nuvens médias, nuvens altas, cirrus e neve/gelo das imagens Sentinel-2.

def mask_sentinel2(image):

    scl = image.select('SCL')              # Seleciona a banda de classificação

    mask = (
        scl.neq(3)      # Remove sombras de nuvens
        .And(scl.neq(8))   # Remove nuvens médias
        .And(scl.neq(9))   # Remove nuvens altas
        .And(scl.neq(10))  # Remove nuvens cirrus
        .And(scl.neq(11))  # Remove neve
    )

    return (
        image
        .updateMask(mask)                  # Aplica a máscara
        .divide(10000)                     # Converte para reflectância
        .copyProperties(image, ['system:time_start'])  # Mantém a data da imagem
    )

Etapa 6 — CALCULAR ÍNDICES — NDVI, NDWI E CMRI

Cria uma função para calcular os índices espectrais NDVI, NDWI e CMRI e adicioná-los como bandas à imagem.

# Função para calcular os índices espectrais
def add_indices(image):

    ndvi = image.normalizedDifference(['B8', 'B4']).rename('NDVI')   # Índice de vegetação

    ndwi = image.normalizedDifference(['B3', 'B8']).rename('NDWI')   # Índice de água

    cmri = ndvi.subtract(ndwi).rename('CMRI')                        # Índice para manguezais

    return image.addBands([
        ndvi,
        ndwi,
        cmri
    ])                                                               # Adiciona os índices à imagem

Etapa 7 — CARREGAR AS IMAGENS SENTINEL-2

Monta a coleção Sentinel-2 Surface Reflectance Harmonized filtrada pela ROI, percentual de nuvens e funções de pré-processamento.

# Cria a coleção Sentinel-2 já preparada para análise
sen2_class = (
    ee.ImageCollection('COPERNICUS/S2_SR_HARMONIZED')   # Coleção Sentinel-2
    .filterBounds(roi)                                  # Filtra pela área de estudo
    .filter(ee.Filter.lt('CLOUDY_PIXEL_PERCENTAGE', 30))# Mantém imagens com menos de 30% de nuvens
    .map(mask_sentinel2)                                # Remove nuvens e sombras
    .map(add_indices)                                   # Calcula NDVI, NDWI e CMRI
)

Etapa 9 — GERAR SÉRIE TEMPORAL DAS IMAGENS

Define uma função única para gerar painéis temporais 2x3 de RGB, NDVI, NDWI, CMRI contínuo ou CMRI com limiar.

# Função para gerar uma série temporal de imagens
def plot_temporal_series(
    collection,
    roi,
    years,
    product_type='RGB',
    title=None,
    threshold_cmri=None
):

    fig, axes = plt.subplots(2, 3, figsize=(14, 8))  # Cria painel 2x3
    axes = axes.flatten()                            # Organiza os eixos em lista

    for i, year in enumerate(years):                 # Percorre cada ano

        start = ee.Date.fromYMD(year, 8, 1)          # Data inicial
        end = ee.Date.fromYMD(year, 10, 30)          # Data final

        img_year = (
            collection
            .filterDate(start, end)                  # Filtra imagens do período
            .median()                                # Gera composição mediana
            .clip(roi)                               # Recorta para a área de estudo
        )

        vis_params = {
            'region': roi,                           # Região da imagem
            'dimensions': 512                        # Tamanho da miniatura em pixels
        }

        if product_type == 'RGB':                    # Composição RGB
            ee_img_to_plot = img_year
            vis_params.update({
                'bands': ['B4', 'B3', 'B2'],
                'min': 0.02,
                'max': 0.30,
                'gamma': 1.2
            })
            plot_title = f'RGB {year}'

        elif product_type == 'NDVI':                 # Índice de vegetação
            ee_img_to_plot = img_year.select('NDVI')
            vis_params.update({
                'min': -0.2,
                'max': 0.8,
                'palette': ['8c510a', 'ffffbf', '006837']
            })
            plot_title = f'NDVI {year}'

        elif product_type == 'NDWI':                 # Índice de água
            ee_img_to_plot = img_year.select('NDWI')
            vis_params.update({
                'min': -0.5,
                'max': 0.5,
                'palette': ['8c510a', 'f7f7f7', '01665e']
            })
            plot_title = f'NDWI {year}'

        elif product_type == 'CMRI':                 # Índice para manguezal
            cmri = img_year.select('CMRI')

            if threshold_cmri is not None:           # Aplica limiar no CMRI
                cmri_mask = cmri.gt(threshold_cmri)
                ee_img_to_plot = cmri_mask.updateMask(cmri_mask)
                vis_params.update({
                    'min': 0,
                    'max': 1,
                    'palette': ['ffffff', '006837']
                })
                plot_title = f'CMRI > {threshold_cmri:.2f} | {year}'

            else:                                    # Exibe CMRI contínuo
                ee_img_to_plot = cmri
                vis_params.update({
                    'min': -1,
                    'max': 1,
                    'palette': ['313695', '74add1', 'ffffff', 'a6d96a', '006837']
                })
                plot_title = f'CMRI {year}'

        else:
            raise ValueError('Produto inválido.')    # Erro para produto inexistente

        url = ee_img_to_plot.getThumbURL(vis_params) # Gera link da imagem
        response = requests.get(url)                 # Baixa a imagem
        img_plot = Image.open(BytesIO(response.content)) # Abre a imagem

        axes[i].imshow(img_plot)                     # Exibe no painel
        axes[i].set_title(plot_title)                # Define título
        axes[i].axis('off')                          # Remove eixos

    if title is None:                                # Define título geral automático
        if product_type == 'CMRI' and threshold_cmri is not None:
            title = f'Série Histórica CMRI com Limiar Sentinel-2 ({years[0]}–{years[-1]})'
        else:
            title = f'Evolução do {product_type} Sentinel-2 ({years[0]}–{years[-1]})'

    plt.suptitle(title, fontsize=16)                 # Adiciona título geral

    if product_type == 'CMRI' and threshold_cmri is not None:
        legend_elements = [
            mpatches.Patch(
                color='#006837',
                label=f'Manguezal / Vegetação - CMRI > {threshold_cmri:.2f}'
            )
        ]

        fig.legend(handles=legend_elements, loc='lower center', ncol=1, fontsize=9)
        plt.tight_layout(rect=[0, 0.08, 1, 1], h_pad=3)

    else:
        plt.tight_layout(rect=[0, 0.03, 1, 1], h_pad=3)

    if threshold_cmri is not None:                   # Nome do arquivo com limiar
        nome_arquivo = f'CMRI_limiar_{threshold_cmri:.2f}_{years[0]}_{years[-1]}.png'
    else:                                            # Nome do arquivo sem limiar
        nome_arquivo = f'{product_type}_{years[0]}_{years[-1]}.png'

    print('Imagem gerada com sucesso!')
    print(f'Nome sugerido para exportação: {nome_arquivo}')

    plt.show()                                       # Mostra a figura

    return fig, nome_arquivo                         # Retorna figura e nome

Etapa 10 — CRIAR MENU INTERATIVO PARA SELECIONAR O PRODUTO

Cria a interface com menu suspenso, barra de limiar para CMRI e botão para gerar o painel temporal automaticamente.

# Cria um menu suspenso (Dropdown) para selecionar o produto
produto_widget = widgets.Dropdown(
    options=[
        ('RGB', 'RGB'),
        ('NDVI', 'NDVI'),
        ('NDWI', 'NDWI'),
        ('CMRI', 'CMRI'),
        ('CMRI com limiar', 'CMRI_LIMIAR')
    ],
    value='RGB',                     # Opção inicial
    description='Produto:'           # Texto exibido ao lado do menu
)

# Cria uma barra deslizante (Slider) para definir o limiar do CMRI
limiar_widget = widgets.FloatSlider(
    value=0.15,                      # Valor inicial
    min=-1,                          # Valor mínimo
    max=1,                           # Valor máximo
    step=0.01,                       # Incremento
    description='Limiar:',
    continuous_update=False          # Atualiza somente ao soltar o controle
)

# Agrupa o slider em um container (VBox)
limiar_box = widgets.VBox([limiar_widget])
limiar_box.layout.display = 'none'   # Oculta inicialmente

# Exibe o slider apenas quando "CMRI com limiar" for selecionado
def mudar_produto(change):

    if change['new'] == 'CMRI_LIMIAR':
        limiar_box.layout.display = 'flex'
    else:
        limiar_box.layout.display = 'none'

# Monitora alterações no menu suspenso
produto_widget.observe(mudar_produto, names='value')

# Cria um botão para gerar a imagem
botao = widgets.Button(
    description='Gerar imagem',
    button_style='success',          # Cor verde
    icon='image'                     # Ícone
)

# Cria uma área para exibir os resultados
saida = widgets.Output()

# Função executada quando o botão é clicado
def gerar_imagem(b):

    global figura, nome_arquivo      # Armazena a figura e o nome do arquivo

    with saida:

        clear_output(wait=True)      # Limpa a saída anterior

        # Gera o CMRI utilizando um limiar
        if produto_widget.value == 'CMRI_LIMIAR':

            figura, nome_arquivo = plot_temporal_series(
                collection=sen2_class,
                roi=roi,
                years=years,
                product_type='CMRI',
                threshold_cmri=limiar_widget.value
            )

        # Gera RGB, NDVI, NDWI ou CMRI contínuo
        else:

            figura, nome_arquivo = plot_temporal_series(
                collection=sen2_class,
                roi=roi,
                years=years,
                product_type=produto_widget.value
            )

# Associa o clique do botão à função
botao.on_click(gerar_imagem)

# Exibe os componentes da interface
display(produto_widget)   # Menu suspenso
display(limiar_box)       # Barra deslizante
display(botao)            # Botão
display(saida)            # Área onde a imagem será exibida
